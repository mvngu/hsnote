\begin{axis}[%%
  axis line style=thick,%%
  axis lines=center,%%
  dotStyle/.style={only marks,mark size=2,mark=*},%%
  enlargelimits=true,%%
  height=8cm,%%
  plotStyle/.style={%%
    domain=0:2*pi,%%
    mark=none,%%
    smooth,%%
    thick%%
  },%%
  width=12cm,%%
  %%
  %% x-axis
  xlabel={\normalsize $x$},%%
  xlabel style={right},%%
  xtick={%%
    0,1.570796,3.141592,4.712388,6.283185%%
  },%%

    0,$\frac{\pi}{2}$,$\pi$,$\frac{3\pi}{2}$,$2\pi$%%
  },%%
  %%
  %% y-axis

%%

\addplot+ [plotStyle]
{4 * sin(deg(x)) + 2};
%%
%%
%% Horizontal line through the number b, which is the midline.
\draw[thin,dashed] (axis cs:\pgfkeysvalueof{/pgfplots/xmin},2) -- (axis cs:\pgfkeysvalueof{/pgfplots/xmax},2);

%%
%% Special points of the function.
\addplot[dotStyle,black,mark color=black] coordinates {
  (0,2)

  (pi/2,6)
