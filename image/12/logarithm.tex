\documentclass{standalone}

\usepackage{mathptmx}
\usepackage{pgfplots}

\begin{document}

\begin{tikzpicture}
\tikzset{%%
  every mark/.append style={scale=1.0},%%
  scale=1.0%%
}

  every axis/.append style={font=\normalsize}%%
}
%%
\begin{axis}[%%
  axis line style=thick,%%
  axis lines=center,%%
  enlargelimits=true,%%
  legend cell align=left,%%
  legend style={at={(axis cs:0.5,4)},anchor=north west},%%
  plotStyle/.style={%%
    domain=0:10,%%
    mark=none,%%
    smooth,%%
    thick%%
  },%%
  %% x axis

  xlabel style=right,%%
  %% y axis
  ylabel={\normalsize $y$},%%
  ylabel style=above%%
]

%%

\addplot+ [plotStyle,black]
{log2(x)};

%%
%%

\addplot+ [plotStyle,blue]
{ln(x)};

\addplot+ [plotStyle,red]
{log10(x)};

\end{axis}
