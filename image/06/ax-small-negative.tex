%% IEEE uses Times Roman font, so we'll default to Times.
%% These three commands make up the entire times.sty package.
\renewcommand{\rmdefault}{ptm}
\renewcommand{\ttdefault}{pcr}
\normalfont\selectfont

%% Graphs of some functions of the form f(x) = ax, where 0 < a <= 1.
\newcommand{\graphFunctions}{%%
  %% The x-axis.
  \draw[axisStyle] (xstart) -- (xend);
  \foreach \x in {-3,-2,-1,1,2,3}{
    \xtick{\x}
  };
  \node at (xend) [right] {$x$};
  %% The y-axis.
  \draw[axisStyle] (ystart) -- (yend);
  \foreach \y in {-3,-2,-1,1,2,3}{
    \ytick{\y}
  }
  \node at (yend) [above] {$y$};
  %% Some functions.
  \draw[lineStyle] (\xlow,-\ylow) -- (\xhigh,-\yhigh);
  \node at (\xhigh,-\yhigh) [right] {$f(x) = -x$};
  \draw[lineStyle] (\xlow,-\xlow/2) -- (\xhigh,-\xhigh/2);
  \node at (\xhigh,-\xhigh/2) [right] {$f(x) = -\frac{1}{2}x$};
  \draw[lineStyle] (\xlow,-\xlow/3) -- (\xhigh,-\xhigh/3);
  \node at (\xhigh,-\xhigh/3) [right] {$f(x) = -\frac{1}{3}x$};
  \draw[lineStyle] (\xlow,-\xlow/10) -- (\xhigh,-\xhigh/10);
  \node at (\xhigh,-\xhigh/10) [right] {$f(x) = -\frac{1}{10}x$};
  %% Reflection about the x-axis of the above functions.
  \draw[reflectionStyle] (\xlow,\ylow) -- (\xhigh,\yhigh);
  \node at (\xhigh,\yhigh) [right] {$f(x) = x$};
  \draw[reflectionStyle] (\xlow,\xlow/2) -- (\xhigh,\xhigh/2);
  \node at (\xhigh,\xhigh/2) [right] {$f(x) = \frac{1}{2}x$};
  \draw[reflectionStyle] (\xlow,\xlow/3) -- (\xhigh,\xhigh/3);
  \node at (\xhigh,\xhigh/3) [right] {$f(x) = \frac{1}{3}x$};
  \draw[reflectionStyle] (\xlow,\xlow/10) -- (\xhigh,\xhigh/10);
  \node at (\xhigh,\xhigh/10) [right] {$f(x) = \frac{1}{10}x$};
}

%% A tick on the x-axis.
%%
%% #1 -- A point on the x-axis.
\newcommand{\xtick}[1]{%%
  \draw[tickStyle] (#1,\tickdx) -- (#1,-\tickdx);
  \node at (#1,-\tickdx) [below] {$#1$};
}

%% A tick on the y-axis.
%%
%% #1 -- A point on the y-axis.
\newcommand{\ytick}[1]{%%
  \draw[tickStyle] (-\tickdx,#1) -- (\tickdx,#1);
  \node at (-\tickdx,#1) [left] {$#1$};
}
